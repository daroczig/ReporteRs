Header rows

  • Header rows are the gray part of the table. A FlexTable can contain contain several header rows, it can also contain no header rows.
  • Header rows are rows preceding body. Theses rows are optional.
  • Header rows are defined by a set of FlexRow and FlexCell objects. Contents and formats are defined with FlexCell objects.

Body content

  • Body content is the blue part of the table. This content is the data.frame or matrix defined with argument data of FlexTable.
  • Each content can be formatted and each content can be replaced by one or more paragraphs of texts (defined with set_of_paragraphs and pot).
  • Content displayed values are the result of applying function format on data values. Use options to modify your settings if necessary (digits, scipen, etc.). The data can also be only composed of character vectors (character vectors are send as they are) that you formatted the way you want before.

Footer rows

  • Footer rows are the light-blue part of the table. A FlexTable can contain contain several footer rows, it can also contain no footer rows.
  • Footer rows are rows preceding body. Theses rows are optional.
  • Footer rows are defined by a set of FlexRow and FlexCell objects. Contents and formats are defined with FlexCell objects.

Usage

FlexTable require mandatory argument data, it can be a data.frame object or a matrix.

To add row.names, use row.names = TRUE. Remember that it increments the number of columns of your table.

To add data column names as headers, use header.columns = TRUE.

Default format properties can also be specified:

  • cell_format: cellProperties object to apply to each cells.
  • par_format: parProperties object to apply to each cells.
  • text_format: textProperties object to apply to each cells.

Below an example. The following dataset will be used:

library( boot )
library( plyr )

data(melanoma)

melanoma$Status = c( "Melanoma", "Alive" , "Non-melanoma" )[melanoma$status]
melanoma$Gender = ifelse( melanoma$sex > 0, "Male", "Female" )
melanoma$Ulceration = ifelse( melanoma$ulcer > 0, "Present", "Absent" )

data = ddply( melanoma, .(Status, Ulceration), summarize
                , n = length( thickness )
                , Mean = mean( thickness, na.rm = T )
                , SD = sd( thickness, na.rm = T )
                , Median = median( thickness, na.rm = T )
                , Min = min( thickness, na.rm = T )
                , Max = max( thickness, na.rm = T )
                , Missing = sum( is.na( thickness ) )
)
row.names( data ) = paste( "id", 1:nrow(data) )

Merging data cell

The following code show usage of this parameter:

myFlexTable = FlexTable( data = data
        , span.columns = c( "Status", "Ulceration" )
        , header.columns = TRUE
        , row.names = FALSE
)

Replace/Insert content

You can replace cell content with function setFlexCellContent.

Function accepts pot and set_of_paragraphs arguments. Therefore you can insert simple text or multiple lines of text.

myFlexTable = FlexTable( data = data[, 1:3]
                , header.columns = TRUE
                , row.names = TRUE
                , cell_format = cellProperties( background.color="white")
                , par_format = parProperties( text.align="right" )
                , text_format = textProperties( color="gray20" )
        )

uke_comment1 = pot("Ukulele", format=textProperties(color="#102E37", font.weight="bold") ) + pot("also named uke"
        , format=textProperties(color = "#2BBBD8", font.weight="bold", vertical.align="superscript") )
uke_comment2 = pot("a small guitar-like instrument", format=textProperties(color="gray", font.style="italic") )
dummy_texts = set_of_paragraphs( uke_comment1, uke_comment2 )
myFlexTable = setFlexCellContent( myFlexTable, 3, 1, dummy_texts )

First define formating properties that will be used later.

# cellProperties for table header cells
headerCellProp = cellProperties(border.bottom.width = 1
        , border.top.width = 0, border.left.width = 0, border.right.width = 0)

# cellProperties for table body cells
dataCellProp = cellProperties(border.bottom.width = 1
        , border.top.width = 0, border.left.width = 0, border.right.width = 0
        , border.bottom.style = "dashed")

# parProperties for table body cells
dataParProp = parProperties( padding = 2, text.align = "right" )


# textProperties for table body cells
dataTextProp = textProperties( )
# textProperties for table header cells
headerTextProp = textProperties( font.weight = "bold" )
# textProperties used to format text as superscript
superScriptTextProp = textProperties(font.weight="bold", vertical.align="superscript")
# a textProperties used to format SD values
fadedText = textProperties(color = "#666666", font.weight="bold")

Create the object.

# create a FlexTable, span rows of columns Status and Gender
# do not print header column names neither row.names
myFlexTable = FlexTable( data = data[ , c( "Status", "Gender", "Ulceration", "n", "Mean") ]
        , header.columns = FALSE, row.names = FALSE
        , span.columns = c("Status", "Gender")
        , cell_format = dataCellProp
        , text_format = dataTextProp
        , par_format = dataParProp
)

Define header rows.

#prepate a text for a first header row
summary_text = pot( "Summary statistics for thickness", format = headerTextProp ) + pot("*", format=superScriptTextProp )

#create a first header row
headerRow = FlexRow()
# span the first 3 cells and leave content empty 
headerRow[1] = FlexCell( pot( "", format = headerTextProp ), cellProp = cellProperties(border.width = 0), colspan=3)
# span the next 2 cells and add 'summary_text' into
headerRow[2] = FlexCell( summary_text, cellProp = headerCellProp, parProp = parProperties( text.align = "center" ), colspan=2 )
#add the header row to myFlexTable 
myFlexTable = addHeaderRow( myFlexTable, headerRow)

#create a second header row and add labels ".names" into
.names = c( "Status", "Gender", "Ulceration", "Count", "Mean ( sd)")
headerRow = FlexRow(.names, textProp = headerTextProp, cellProp = headerCellProp, parProp = dataParProp )
#add the header row to myFlexTable 
myFlexTable = addHeaderRow( myFlexTable, headerRow)

Define footer rows.

#create a first footer row
footerRow = FlexRow()
#prepate a text for a first footer row
footernote = pot("*", format=superScriptTextProp ) + pot(" Also known as Breslow thickness"
                        , format=fadedText )
# span all cells and add footernote into 
footerRow[1] = FlexCell( footernote
                , parProp = parProperties(text.align="left")
                , cellProp = headerCellProp
                , colspan=5
)
#add the footer row to myFlexTable 
myFlexTable = addFooterRow( myFlexTable, footerRow )

Customize cell contents.

#format mean with 2 digits
#format sd with 2 digits and change it to (SD)
.mean = formatC( data$Mean, digits=2, format = "f")
.sd = paste( " (", formatC( data$SD, digits=2, format = "f"), ")", sep = "" )

for(i in 1:nrow(data) ){
        # create the formated text as a concatenate of mean and sd values
        textVal = pot(.mean[i], format = dataTextProp  ) + pot(.sd[i], format=fadedText )
        # replace line i and column 5 of the table by the formated text
        myFlexTable = setFlexCellContent( myFlexTable, i, 5, textVal )
}

        , border.left.width = 0, border.right.width = 0
